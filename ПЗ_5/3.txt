#include <iostream>
using namespace std;

int main()
{
	cout << "exercise 3:\n";
	int a, b, h;
	int y = 0;

	cout << "Enter a:";
	cin >> a;

	cout << "Enter b:";
	cin >> b;

	cout << "Enter h:";
	cin >> h;

	for (int x = a; x <= b; x += h) {
		if (x + 5 != 0) {
			y = 8 + (sqrt(x - 7)) / (x + 5);
			cout << "-----------------------:\n";
			cout << ":   X :        Y       :\n";
			cout << "-----------------------:\n";
			cout << ":   " << x << " : " << y << "      :\n";
			cout << "-----------------------:\n";
		}
		else {
			cout << "Invalid value for x";
		}
	}
	return 0;
	
}
